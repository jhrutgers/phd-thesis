\def\getword#1 {%
	\let\donext\getword%
	\ifx\relax#1%
		\let\donext\relax%
	\else%
		\expandafter\appto\expandafter\authorfirstnames\expandafter{\expandafter\space\authorlastname}%
		\def\authorlastname{#1}%
	\fi%
	\donext%
}
